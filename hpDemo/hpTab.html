动态选项卡
hpTab 是建立 在 layui 的 自定义 模块上的,所以使用起来和layui依赖自身的模块是一样的 
//layui模块的使用
			layui.use(['mod1', 'mod2'], function(args){
  		    var mod = layui.mod1;
            //……
 });   
		
// 所以使用 hptab 如下 
			// 配置
			layui.config({
				base: './hpModules/' // 扩展模块目录
			}).extend({ // 模块别名
				hpTab: 'hpComponent/hpTab',
			});
			layui.use(['layer','hpTab'], function(args){
  		    var hpTab = layui.hpTab;
            hpTab.init();
            //……
            
 });
